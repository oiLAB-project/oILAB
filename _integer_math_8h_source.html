136

143  tempx(i + 1) = 0;

148  Eigen::Vector<IntScalarType,Eigen::Dynamic>::Zero(i+2);

152  std::vector<IntScalarType> indicesToKeep(i+2);

181  // Finds such x and y, that a x + b y = gcd(a, b)
182  //https://github.com/ADJA/algos/blob/master/NumberTheory/DiophantineEquation.cpp
183  static IntScalarType extended_gcd(IntScalarType a, IntScalarType b, IntScalarType &x, IntScalarType &y)
184  {
185  if (b == 0)
186  {
187  x = 1;
188  y = 0;
189  return a;
190  }
191  IntScalarType x1, y1;
192  IntScalarType g = extended_gcd(b, a % b, x1, y1);
193  x = y1;
194  y = x1 - (a / b) * y1;
195  return g;
196  }

217  // Find u such that a_1 u_1 + a_2 u_2 + .... + a_n u_n = 1
218  // Method 0:
219  // "A fast algorithm to find reduced hyperplane unit cells and solve N-dimensional Bézout's identities."
220  // Acta Crystallographica Section A: Foundations and Advances 77.5 (2021).
221  // template<typename ArrayType>
222  // static ArrayType solveBezout(const ArrayType& a)
223  template<typename T>
224  static Eigen::Vector<IntScalarType,Eigen::Dynamic> solveBezout(const Eigen::MatrixBase<T>& a)
225  {
226  int n= a.size();
227  if (n<2) throw std::runtime_error("the size of arrays should be at least two");
228  if (abs(IntegerMath<IntScalarType>::gcd(a)) != 1 || a.isZero())
229  throw std::runtime_error("No solution since the gcd is not 1 or -1.");
230 
231  Eigen::Vector<IntScalarType,Eigen::Dynamic> u(n);
232 
233  IntScalarType p,q;
234  IntScalarType g12= extended_gcd(a(0),a(1),p,q); // g12 - gcd of a(0) and a(1)
235 
236  // form the n-1 sized vector na= {g,a2,...,a_{n-1}}
237  Eigen::Vector<IntScalarType,Eigen::Dynamic> na(n-1);
238  na(0)= g12; na(Eigen::seq(1,n-2))= a(Eigen::seq(2,n-1));
239 
240  Eigen::Vector<IntScalarType,Eigen::Dynamic> k(n-1);
241  if (n==2)
242  {
243  IntScalarType g= extended_gcd(a(0),a(1),u(0),u(1));
244  if (g<0) u= -u;
245  return u;
246  }
247  else
248  {
249  k = solveBezout(na);
250  // {p*k0,q*k0,k1,..,k_{n-2}}
251  u(0) = p * k(0);
252  u(1) = q * k(0);
253  u(Eigen::seq(2, n - 1)) = k(Eigen::seq(1, n - 2));
254  return u;
255  }
