53 void perform_op(const double *x_in, double *y_out) const {
54 int nx = this->rows();
55 Eigen::VectorXd y1(nx);
56 Eigen::VectorXd y2(nx);
57 o1.perform_op(x_in, y1.data());
58 o2.perform_op(x_in, y2.data());
59 Eigen::Map<Eigen::VectorXd> y(y_out, nx);
