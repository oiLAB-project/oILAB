127  generateCoincidentLattices(const ReciprocalLatticeDirection<dim>& rd, const double& maxDen= 100, const int N= 100) const
128  {
129  std::vector<MatrixDimD> output;

131  auto basis= planeParallelLatticeBasis(rd);
132  double epsilon=1e-8;
133  IntScalarType keyScale= 1e6;
134 
135  auto b1= basis[1].cartesian();
136  auto b2= basis[2].cartesian();
137 
138  // Following the notation in Algorithm 3 of
139  // "Interface dislocations and grain boundary disconnections using Smith normal bicrystallography"
140  auto q1= b1;
141  std::vector<std::pair<int,int>> coPrimePairs= farey(N,false);
142  for(const auto& pair : coPrimePairs)
143  {
144  VectorDimI pIndices;
145  auto q2= pair.first*b1+pair.second*b2;
146 
147  if (abs(q2.norm())<epsilon ) continue;
148  double ratio= q2.norm()/q1.norm();
149  BestRationalApproximation bra(ratio,maxDen);
150  double error= ratio-static_cast<double>(bra.num)/bra.den;
151  if (abs(error) > epsilon)
152  continue;
153  else
154  {
155  double cosTheta= b1.dot(q2)/(q1.norm()*q2.norm());
156  if (cosTheta-1>-epsilon) cosTheta= 1.0;
157  if (cosTheta+1<epsilon) cosTheta= -1.0;
158  double theta= acos(cosTheta);
159  MatrixDimD rotation;
160  rotation= Eigen::AngleAxis<double>(theta,rd.cartesian().normalized());
161  IntScalarType key= theta*keyScale;
162  temp.insert(std::pair<IntScalarType,MatrixDimD>(key,rotation));
163  }
164  }
165  std::transform(temp.begin(), temp.end(),
166  std::back_inserter(output),
167  [](const std::pair<IntScalarType,MatrixDimD>& p) {
168  return p.second;
169  });

206  typename std::enable_if<dm==2,std::vector<MatrixDimD>>::type
207  generateCoincidentLattices(const Lattice<dim>& undeformedLattice,
208  const double& maxStrain,
209  const int& maxDen=50,
210  const int& N=30) const
211  {
212  std::vector<MatrixDimD> output;
213  std::map<IntScalarType,MatrixDimD> temp;
214  MatrixDimI mn(MatrixDimI::Zero());
215  MatrixDimI md(MatrixDimI::Ones());
216  std::vector<std::pair<int,int>> coPrimePairs= farey(N,false);
217  double epsilon=1e-8;
218  IntScalarType keyScale= 1e6;
219 
220  for(const auto& pair1 : coPrimePairs)
221  {
222  VectorDimI pIndices;
223  pIndices << pair1.first, pair1.second;
224  LatticeVector<dm> q2(pIndices,undeformedLattice);
225  double ratio= q2.cartesian().norm() / latticeBasis.col(0).norm();
226 
227 
228  if (abs(maxStrain) < epsilon)
229  {
230  BestRationalApproximation alpha(ratio, maxDen);
231  double error = ratio - static_cast<double>(alpha.num) / alpha.den;
232  if (abs(error) > epsilon)
233  continue;
234  else
235  {
236  double cosTheta = latticeBasis.col(0).dot(q2.cartesian()) / (latticeBasis.col(0).norm() * q2.cartesian().norm());
237  if (cosTheta - 1 > -epsilon) cosTheta = 1.0;
238  if (cosTheta + 1 < epsilon) cosTheta = -1.0;
239  double theta = acos(cosTheta);
240  Eigen::Rotation2D<double> rotation(theta);
241  IntScalarType key= theta*keyScale;
242  temp.insert(std::pair<IntScalarType,MatrixDimD>(key,rotation.toRotationMatrix()));
243  }
244  }
245  else
246  {
247  RationalApproximations <IntScalarType> alphaSequence(ratio, maxDen, ratio*maxStrain);
248  for (const auto& alpha: alphaSequence.approximations)
249  {
250  RationalLatticeDirection<dm> q2ByAlpha(Rational<IntScalarType>(alpha.d, alpha.n), q2);
251 
252  mn.col(0) = q2 * alpha.d;
253  md.col(0).setConstant(alpha.n);
254 
255  double s1 =
256  (q2ByAlpha.cartesian().squaredNorm() - latticeBasis.col(0).squaredNorm()) /
257  latticeBasis.col(0).squaredNorm();
258  if (abs(s1) > maxStrain) continue;
259 
260  for (const auto &pair2: coPrimePairs) {
261  VectorDimI qIndices;
262  qIndices << pair2.first, pair2.second;
263  LatticeVector<dm> r2(qIndices, undeformedLattice);
264  double ratio2= r2.cartesian().norm() / latticeBasis.col(1).norm();
265  RationalApproximations<IntScalarType> betaSequence(ratio2, maxDen,maxStrain*ratio2);
266  for(const auto& beta : betaSequence.approximations) {
267  RationalLatticeDirection<dm> r2ByBeta(Rational<IntScalarType>(beta.d, beta.n), r2);
268  double s2 = (r2ByBeta.cartesian().squaredNorm() - latticeBasis.col(1).squaredNorm()) /
269  latticeBasis.col(1).squaredNorm();
270  double s3 = (q2ByAlpha.cartesian().dot(r2ByBeta.cartesian()) -
271  latticeBasis.col(0).dot(latticeBasis.col(1))) /
272  (latticeBasis.col(0).norm() * latticeBasis.col(1).norm());
273  if (abs(s2) > maxStrain || abs(s3) > maxStrain) continue;
274 
275  // calculate the deformation gradient
276  MatrixDimD F = q2ByAlpha.cartesian() * reciprocalBasis.col(0).transpose() +
277  r2ByBeta.cartesian() * reciprocalBasis.col(1).transpose();
278  if (F.determinant() < 0) continue;
279 
280  output.push_back( F );
281  }
282  }
283  }
284  }
285  }
286  // sort the angles if rotations are being asked
287  if (abs(maxStrain) < epsilon)
288  std::transform(temp.begin(), temp.end(),
289  std::back_inserter(output),
290  [](const std::pair<IntScalarType,MatrixDimD>& p) {
291  return p.second;
292  });
293  return output;
294  }
