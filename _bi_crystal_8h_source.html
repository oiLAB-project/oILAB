199  generateGrainBoundaries(const LatticeDirection<dim>& d, int div=30) const
200  {
201  if (&d.lattice != &A && &d.lattice != &B)
202  throw std::runtime_error("The tilt axis does not belong to lattices A and B \n");
203  std::vector<Gb<dm>> gbVec;
204  double epsilon=1e-8;
205  int count= -1;
206  IntScalarType keyScale= 1e6;
207  auto basis= d.lattice.directionOrthogonalReciprocalLatticeBasis(d,true);
208  if (dm==3)
209  {
210  for (int i = -div; i <= div; ++i)
211  {
212  for (int j = -div; j <= div; ++j)
213  {
214  if (i==0 && j==0) continue;
215  count++;
216  ReciprocalLatticeVector<dm> rv = i * basis[1].reciprocalLatticeVector() + j * basis[2].reciprocalLatticeVector();
217  try
218  {
219  Gb<dm> gb(*this, rv);
220  gbVec.push_back(gb);
221  }
222  catch(std::runtime_error& e)
223  {
224  std::cout << e.what() << std::endl;
225  std::cout << "Unable to form GB with normal = " << rv << std::endl;
226  std::cout << "moving on to next inclination" << std::endl;
227  }
228  }
229  }
230  }
231  else if(dm==2)
232  {
233  auto rv= basis[0].reciprocalLatticeVector();
234  gbVec.push_back(Gb<dm>(*this, rv));
235  }
236  std::map<IntScalarType,Gb<dm>> gbSet;
237  for(const Gb<dim>& gb:gbVec)
238  {
239  double cosAngle;
240  cosAngle= gb.nA.cartesian().normalized().dot(gbVec[0].nA.cartesian().normalized());
241  if (cosAngle-1>-epsilon) cosAngle= 1.0;
242  if (cosAngle+1<epsilon) cosAngle= -1.0;
243 
244  double angle= acos(cosAngle);
245  IntScalarType key= angle*keyScale;
246  gbSet.insert(std::pair<IntScalarType,Gb<dm>>(key,gb));
247  }
